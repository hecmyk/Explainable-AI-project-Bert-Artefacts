"""Phase C — DOM builder.

build_dom is a PURE function of (doc_ocr, page_dims): no I/O, no PDF, no network.
The runner (lot 7) computes page_dims via pdf_utils.page_dims_from_pdf(pdf_path) and
calls build_dom, then serializes the result to doc_dom.json next to doc_ocr.json.

Scope so far:
  lot 1 — document header + flat index (content/visual atoms) + per-page chrome bucket.
  lot 2 — section tree + section_id backpointer on each atom, caption resolution
          (figure_title/footnote adjacent — previous or next — to a visual on the same
          page), zero-title per-page fallback.
  lot 3 — formal ToC detection (header.toc_present / toc_atom_ids).
  lot 4 — title levels (1-4) via an INJECTED title_leveler, then the flat section list is
          nested by level. Without a leveler every section stays L1 (lot-2 shape).

Not yet (later lots):
  audit sidecar ................................... later
  running-head de-dup (KNOWN lot-2 limitation: a paragraph_title repeated identically
      across pages opens a phantom section each time — accepted for now) ... dedicated lot
"""

# Union of the three role sets = the labels we know how to route. The sets are disjoint
# and cover the 25 PP-DocLayoutV2 labels exactly once; an unknown label makes _classify
# raise (v1 is expected never to emit one).
_KNOWN_LABELS = DOM_CONTENT_LABELS | DOM_VISUAL_LABELS | DOM_CHROME_LABELS


def _classify(label):
    """(role, visual_kind) for a label. The three role sets are disjoint; for a
    visual atom the visual_kind IS the label (table/chart/figure/image)."""
    if label in DOM_VISUAL_LABELS:
        return "visual", label
    if label in DOM_CHROME_LABELS:
        return "chrome", None
    if label in DOM_CONTENT_LABELS:
        return "content", None
    raise ValueError(f"Unknown label: {label!r}")


def _ordered_stream(doc_ocr):
    """Flat list of every atom in global reading order (page asc, then dim7 asc).

    Order comes straight from the iteration: pages are visited in sorted key order and
    each doc_ocr[page_key] is already in dim7 order (Phase B invariant), so no extra sort
    is needed. The atom_id is assigned from the ORIGINAL per-page list index (enumerate),
    so it matches markdown_build/id_to_bbox unconditionally (see [[dom-id-parity]]). Each
    record carries the routed role/visual_kind so caption detection and grouping don't
    reclassify."""
    stream = []
    for page_key in sorted(doc_ocr, key=lambda k: int(k.split("_")[1])):
        page = int(page_key.split("_")[1])
        for idx, item in enumerate(doc_ocr[page_key]):
            role, visual_kind = _classify(item["label"])
            stream.append({
                "atom_id": f"p{page}_e{idx}",
                "page": page,
                "dim7": item["dim7"],
                "item": item,
                "label": item["label"],
                "role": role,
                "visual_kind": visual_kind,
            })
    return stream


def _detect_captions(stream):
    """{caption_atom_id: visual_atom_id}. A figure_title/footnote atom is a caption
    iff its PREVIOUS or NEXT content-or-visual atom on the SAME page is either a
    visual OR an already-resolved caption of a visual (chain V -> C1 -> C2).
    Previous wins over next when both match."""
    adj = [a for a in stream
           if a["role"] != "chrome" or a["label"] in DOM_CAPTION_LABELS]
    caption_of = {}
    for i, a in enumerate(adj):
        if i == 0 or a["label"] not in DOM_CAPTION_LABELS:
            continue

        # Previous neighbour (priority)
        prev = adj[i - 1]
        if prev["page"] == a["page"]:
            if prev["role"] == "visual":
                caption_of[a["atom_id"]] = prev["atom_id"]
                continue
            if prev["atom_id"] in caption_of:
                caption_of[a["atom_id"]] = caption_of[prev["atom_id"]]
                continue

        # Fallback: next neighbour
        if i + 1 < len(adj):
            nxt = adj[i + 1]
            if nxt["page"] == a["page"] and nxt["role"] == "visual":
                caption_of[a["atom_id"]] = nxt["atom_id"]

    return caption_of


def _detect_toc(stream):
    """Any content atom on pages 0-4 flags the doc as having a ToC."""
    toc_ids = []
    for a in stream:
        if a["page"] > 4:
            break
        if a["label"] == "content":
            toc_ids.append(a["atom_id"])
    return toc_ids


def _route_atoms(stream, caption_of):
    """Build (index, chrome). content/visual atoms and captions go to the index (a
    caption footnote is PROMOTED from chrome to the index). Non-caption chrome atoms go
    to the per-page chrome bucket. Insertion order = global reading order."""
    index = {}
    chrome = {}
    for a in stream:
        item, atom_id = a["item"], a["atom_id"]
        is_caption = atom_id in caption_of

        if a["role"] == "chrome" and not is_caption:
            chrome.setdefault(str(a["page"]), []).append({
                "id": atom_id,
                "page": a["page"],
                "bbox": item["box"],
                "label": a["label"],
                "text": item.get("text", ""),
                "role": "chrome",
            })
            continue

        atom = {
            "id": atom_id,
            "page": a["page"],
            "bbox": item["box"],
            "label": a["label"],
            "text": item.get("text", ""),
            "ocr_error": item.get("ocr_error"),
            "dim7": a["dim7"],
            "role": "visual" if a["role"] == "visual" else "content",
            "visual_kind": a["visual_kind"],
            "section_id": None,   # filled by _build_sections
        }
        if is_caption:
            atom["caption_of"] = caption_of[atom_id]
        index[atom_id] = atom
    return index, chrome


def _new_section(section_id, title, title_id):
    return {
        "section_id": section_id,
        "title": title,
        "title_id": title_id,
        "level": 1,
        "level_source": "fallback",   # "llm" once a leveler overrides it (lot 4)
        "page_start": None,           # filled by _fill_section_pages
        "page_end": None,
        "children": [],               # atom_ids — NOT sub-sections (those go to "subsections")
        "subsections": [],            # child sections, filled by _nest_sections (lot 4)
    }


def _build_sections(stream, index, caption_of):
    """Walk the index atoms in reading order and group them into L1 sections.

    A title atom (doc_title/paragraph_title) opens a section unless it is a caption. A
    figure_title never opens a section: an orphan figure_title (no visual to caption)
    attaches to the current section as normal content. Every other atom attaches to the
    current section; atoms appearing before any section is open land in a lazily-created
    implicit holder 's_toc'. Fills section_id on each index atom.

    Returns (sections, opened_real, first_doc_title) where opened_real counts sections
    opened by a real title (s_toc excluded) and first_doc_title is (atom_id, title) of the
    first doc_title in reading order, or None."""
    sections = []
    current = None
    next_sid = 0
    opened_real = 0
    first_doc_title = None

    def attach(atom_id):
        nonlocal current
        if current is None:
            current = _new_section("s_toc", None, None)
            sections.append(current)
        current["children"].append(atom_id)
        index[atom_id]["section_id"] = current["section_id"]

    for a in stream:
        atom_id = a["atom_id"]
        if atom_id not in index:
            continue  # non-caption chrome: not part of any section

        label = a["label"]
        text = (a["item"].get("text") or "").strip()

        if label == "doc_title" and first_doc_title is None:
            first_doc_title = (atom_id, text or None)

        is_caption = atom_id in caption_of
        is_opener = (not is_caption) and label in {"doc_title", "paragraph_title"}

        if is_opener:
            sid = f"s{next_sid}"
            next_sid += 1
            current = _new_section(sid, text or None, atom_id)
            sections.append(current)
            current["children"].append(atom_id)
            index[atom_id]["section_id"] = sid
            opened_real += 1
        else:
            attach(atom_id)

    return sections, opened_real, first_doc_title


def _sections_per_page(stream, index):
    """Zero-title fallback: one L1 section per page (page_{n}), holding that page's index
    atoms in reading order. Overwrites section_id on every index atom."""
    sections = []
    by_page = {}
    for a in stream:
        if a["atom_id"] in index:
            by_page.setdefault(a["page"], []).append(a["atom_id"])
    for page in sorted(by_page):
        sid = f"page_{page}"
        for cid in by_page[page]:
            index[cid]["section_id"] = sid
        sec = _new_section(sid, None, None)
        sec["children"] = by_page[page]
        sections.append(sec)
    return sections


def _fill_section_pages(sections, index):
    """page_start/page_end from the section's OWN atoms ("children"), EXCLUDING its
    sub-sections — a parent section does not span its subsections (the nested XML skeleton
    already expresses containment). min/max is defensive."""
    for sec in sections:
        pages = [index[cid]["page"] for cid in sec["children"]]
        sec["page_start"] = min(pages) if pages else None
        sec["page_end"] = max(pages) if pages else None


def _apply_title_levels(sections, index, title_leveler):
    """Set level/level_source on every section; return the header title_leveling flag.

    Sections default to level 1 / "fallback" (from _new_section). If a leveler is given
    AND there is at least one titled section (title_id set, title non-empty), the leveler
    is called ONCE with [{title_id, text, page}, ...] in reading order. A returned pair is
    kept only when its title_id is known and its level is an int in 1..4; those sections
    switch to that level / "llm". Everything else stays L1 / "fallback".

    Returns "skipped" (no leveler or no titled section — no call made), "llm" (leveler
    produced at least one usable level), or "failed" (leveler returned None OR produced no
    usable level — the silent-failure case this flag exists to catch)."""
    titled = [s for s in sections if s["title_id"] and s["title"]]
    if title_leveler is None or not titled:
        return "skipped"

    titles = [{"title_id": s["title_id"],
               "text": s["title"],
               "page": index[s["title_id"]]["page"]} for s in titled]
    raw = title_leveler(titles)
    if raw is None:
        return "failed"

    input_ids = {s["title_id"] for s in titled}
    levels = {tid: lvl for tid, lvl in raw.items()
              if tid in input_ids and isinstance(lvl, int)
              and not isinstance(lvl, bool) and 1 <= lvl <= 4}
    for s in titled:
        if s["title_id"] in levels:
            s["level"] = levels[s["title_id"]]
            s["level_source"] = "llm"
    return "llm" if levels else "failed"


def _nest_sections(sections):
    """Nest the flat, reading-ordered section list by level; return the top-level ones.

    Monotonic-stack nesting: a section of level N is appended to the "subsections" of the
    nearest preceding still-open section of level < N; with none open it stays top-level.
    With every section at L1 (no leveler) this returns the flat list unchanged and leaves
    all "subsections" empty (lot-2 shape preserved)."""
    top = []
    stack = []
    for sec in sections:
        while stack and stack[-1]["level"] >= sec["level"]:
            stack.pop()
        if stack:
            stack[-1]["subsections"].append(sec)
        else:
            top.append(sec)
        stack.append(sec)
    return top


def _assemble_tree(top_sections, page_count, first_doc_title):
    """L0 root holding the top-level sections of the nested tree. Root title = first
    doc_title (or null). The key is "sections" (not "children") on purpose: a section's
    "children" are atom_ids, the root's "sections" are section objects."""
    return {
        "section_id": "root",
        "title": first_doc_title[1] if first_doc_title else None,
        "title_id": first_doc_title[0] if first_doc_title else None,
        "level": 0,
        "page_start": 0,
        "page_end": max(page_count - 1, 0),
        "sections": top_sections,
    }


def build_dom(doc_ocr, page_dims, doc_id=None, pdf_name=None, dpi=DPI, title_leveler=None):
    """doc_ocr {page_N: [{label, box, dim7, text, ocr_error, ...}, ...]} +
    page_dims {page_index: (width_px, height_px)} -> doc_dom dict {header, index, chrome, tree}.

    title_leveler: optional callable [{title_id, text, page}, ...] -> {title_id: level} (or
    None on failure). When given, it is called ONCE to assign title levels 1-4 and the flat
    section list is nested by level; when None, every section stays L1. NOTE: the injected
    callable MAY do network I/O — build_dom itself stays pure, the effect is the caller's."""
    stream = _ordered_stream(doc_ocr)
    caption_of = _detect_captions(stream)
    index, chrome = _route_atoms(stream, caption_of)
    toc_atom_ids = _detect_toc(stream)

    sections, opened_real, first_doc_title = _build_sections(stream, index, caption_of)
    if opened_real == 0:
        sections = _sections_per_page(stream, index)
    _fill_section_pages(sections, index)

    title_leveling = _apply_title_levels(sections, index, title_leveler)
    n_sections = len(sections)                # flat total, before nesting
    top_sections = _nest_sections(sections)

    page_count = len(page_dims)
    tree = _assemble_tree(top_sections, page_count, first_doc_title)

    pages = [{"index": p, "width": w, "height": h}
             for p, (w, h) in sorted(page_dims.items())]
    n_visual = sum(1 for a in index.values() if a["role"] == "visual")
    n_chrome = sum(len(v) for v in chrome.values())

    header = {
        "doc_id": doc_id,
        "schema_version": SCHEMA_VERSION,
        "source": {"doc_ocr": "doc_ocr.json", "pdf": pdf_name},
        "dpi": dpi,
        "coord_space": "raster_px",
        "page_count": page_count,
        "pages": pages,
        "toc_present": bool(toc_atom_ids),
        "toc_atom_ids": toc_atom_ids,
        "title_leveling": title_leveling,
        "stats": {
            "n_atoms": len(index),      # content + visual + captions (chrome excluded)
            "n_sections": n_sections,
            "n_chrome": n_chrome,
            "n_visual": n_visual,
        },
    }

    return {"header": header, "index": index, "chrome": chrome, "tree": tree}
