SCHEMA_VERSION = "1.0.0"

# Union of the three role sets = the labels we know how to route. Anything outside
# is an unknown label (should not happen in v1); it falls back to the content role
# so the atom stays visible to the agent instead of being silently dropped.
_KNOWN_LABELS = DOM_CONTENT_LABELS | DOM_VISUAL_LABELS | DOM_CHROME_LABELS

# ToC tolerance (lot 2): on the first pages a TITLE whose text carries dot-leaders
# followed by a number is a table-of-contents entry, not a section opener. Loose,
# single-match, anywhere in the text.
_TOC_RE = re.compile(r"\.{2,}\s*\d+")
_TOC_MAX_PAGE = 2

# ToC detection (lot 3): a stricter, line-anchored pattern for flagging content/table
# atoms that hold an actual table of contents. Dot-leaders (>=2 dots or the unicode
# ellipsis) then a trailing number at end of line.
_TOC_DOTS_NUM = re.compile(r"(?:\.{2,}|…)\s*\d+\s*$")

_UNTITLED = "[untitled]"


def _classify(label):
    """(role, visual_kind) for a label. The three role sets are disjoint; for a
    visual atom the visual_kind IS the label (table/chart/figure/image)."""
    if label in DOM_VISUAL_LABELS:
        return "visual", label
    if label in DOM_CHROME_LABELS:
        return "chrome", None
    # content labels + unknown-label fallback (kept as content, see note above)
    return "content", None


def _ordered_stream(doc_ocr):
    """Flat list of every atom in global reading order (page asc, then dim7 asc).

    The atom_id is assigned from the ORIGINAL per-page list index (enumerate) BEFORE
    sorting, so it matches markdown_build/id_to_bbox unconditionally; the sort only
    fixes the processing order (see [[dom-id-parity]]). Each record carries the routed
    role/visual_kind so caption detection and grouping don't reclassify."""
    stream = []
    for page_key in sorted(doc_ocr, key=lambda k: int(k.split("_")[1])):
        page = int(page_key.split("_")[1])
        for idx, item in enumerate(doc_ocr[page_key]):
            role, visual_kind = _classify(item["label"])
            stream.append({
                "atom_id": f"p{page}_e{idx}",
                "page": page,
                "dim7": item["dim7"],
                "item": item,
                "label": item["label"],
                "role": role,
                "visual_kind": visual_kind,
            })
    stream.sort(key=lambda a: (a["page"], a["dim7"]))
    return stream


def _detect_captions(stream):
    """{caption_atom_id: visual_atom_id}. A figure_title/footnote atom is a caption iff
    the preceding CONTENT-OR-VISUAL atom on the SAME page (chrome ignored — adjustment B)
    is either a visual OR an already-resolved caption (adjustment A). In the caption case
    it inherits that caption's visual, so a chain V -> C1 -> C2 all point to V. Chrome
    atoms (page numbers, running heads) never break the visual->caption adjacency."""
    # Adjacency stream: content + visual + caption candidates. Footnotes are chrome-role
    # but must participate as candidates; pure chrome (header/number/...) is excluded.
    adj = [a for a in stream
           if a["role"] != "chrome" or a["label"] in DOM_CAPTION_LABELS]
    caption_of = {}
    for i, a in enumerate(adj):
        if i == 0 or a["label"] not in DOM_CAPTION_LABELS:
            continue
        prev = adj[i - 1]
        if prev["page"] != a["page"]:
            continue
        if prev["role"] == "visual":
            caption_of[a["atom_id"]] = prev["atom_id"]
        elif prev["atom_id"] in caption_of:
            caption_of[a["atom_id"]] = caption_of[prev["atom_id"]]
    return caption_of


def _is_toc_line(line):
    """A ToC entry line: >=3 non-whitespace chars, then dot-leaders (>=2 dots or the
    unicode ellipsis), then a trailing number at end of line."""
    m = _TOC_DOTS_NUM.search(line)
    if not m:
        return False
    return sum(not c.isspace() for c in line[:m.start()]) >= 3


def _detect_toc(stream):
    """atom_ids (reading order) of content/table atoms on pages 0-2 whose text holds >=3
    distinct ToC-entry lines. Metadata only: the atoms stay normal in the index, they are
    just flagged. Strictly limited to pages 0-2 (the pattern can appear by coincidence
    elsewhere). Returns [] when no ToC is found."""
    toc_ids = []
    for a in stream:
        if a["page"] > _TOC_MAX_PAGE or a["label"] not in ("content", "table"):
            continue
        text = a["item"].get("text") or ""
        matched = {ln.strip() for ln in text.split("\n") if _is_toc_line(ln)}
        if len(matched) >= 3:
            toc_ids.append(a["atom_id"])
    return toc_ids


def _route_atoms(stream, caption_of):
    """Build (index, chrome). content/visual atoms and captions go to the index (a
    caption footnote is PROMOTED from chrome to the index). Non-caption chrome atoms go
    to the per-page chrome bucket. Insertion order = global reading order."""
    index = {}
    chrome = {}
    for a in stream:
        item, atom_id = a["item"], a["atom_id"]
        is_caption = atom_id in caption_of

        if a["role"] == "chrome" and not is_caption:
            chrome.setdefault(str(a["page"]), []).append({
                "id": atom_id,
                "page": a["page"],
                "bbox": item["box"],
                "label": a["label"],
                "text": item.get("text", ""),
                "role": "chrome",
            })
            continue

        atom = {
            "id": atom_id,
            "page": a["page"],
            "bbox": item["box"],
            "label": a["label"],
            "text": item.get("text", ""),
            "ocr_error": item.get("ocr_error"),
            "dim7": a["dim7"],
            "role": "visual" if a["role"] == "visual" else "content",
            "visual_kind": a["visual_kind"],
            "section_id": None,   # filled by _build_sections
        }
        if is_caption:
            atom["caption_of"] = caption_of[atom_id]
        index[atom_id] = atom
    return index, chrome


def _new_section(section_id, title, title_id):
    return {
        "section_id": section_id,
        "title": title,
        "title_id": title_id,
        "level": 1,
        "page_start": None,   # filled by _fill_section_pages
        "page_end": None,
        "children": [],
        "continues_from": None,   # v2
        "continues_to": None,     # v2
    }


def _build_sections(stream, index, caption_of):
    """Walk the index atoms in reading order and group them into L1 sections.

    A title atom (doc_title/paragraph_title/figure_title) opens a section unless it is a
    caption or a ToC entry on pages 0-2. Every other atom attaches to the current
    section; atoms appearing before any section is open land in a lazily-created implicit
    holder 's_toc'. Fills section_id on each index atom.

    Returns (sections, opened_real, first_doc_title) where opened_real counts sections
    opened by a real title (s_toc excluded) and first_doc_title is (atom_id, title) of the
    first doc_title in reading order, or None."""
    sections = []
    current = None
    next_sid = 0
    opened_real = 0
    first_doc_title = None

    def attach(atom_id):
        nonlocal current
        if current is None:
            current = _new_section("s_toc", None, None)
            sections.append(current)
        current["children"].append(atom_id)
        index[atom_id]["section_id"] = current["section_id"]

    for a in stream:
        atom_id = a["atom_id"]
        if atom_id not in index:
            continue  # non-caption chrome: not part of any section

        label = a["label"]
        raw_text = a["item"].get("text") or ""
        text = raw_text.strip()

        if label == "doc_title" and first_doc_title is None:
            first_doc_title = (atom_id, text if text else _UNTITLED)

        is_caption = atom_id in caption_of
        toc_matched = a["page"] <= _TOC_MAX_PAGE and bool(_TOC_RE.search(raw_text))
        is_opener = (not is_caption) and (label in DOM_TITLE_LABELS) and (not toc_matched)

        if is_opener:
            sid = f"s{next_sid}"
            next_sid += 1
            current = _new_section(sid, text if text else _UNTITLED, atom_id)
            sections.append(current)
            current["children"].append(atom_id)
            index[atom_id]["section_id"] = sid
            opened_real += 1
        else:
            attach(atom_id)

    return sections, opened_real, first_doc_title


def _sections_per_page(stream, index):
    """Zero-title fallback: one L1 section per page (page_{n}), holding that page's index
    atoms in reading order. Overwrites section_id on every index atom."""
    sections = []
    by_page = {}
    for a in stream:
        if a["atom_id"] in index:
            by_page.setdefault(a["page"], []).append(a["atom_id"])
    for page in sorted(by_page):
        sid = f"page_{page}"
        for cid in by_page[page]:
            index[cid]["section_id"] = sid
        sec = _new_section(sid, None, None)
        sec["children"] = by_page[page]
        sections.append(sec)
    return sections


def _fill_section_pages(sections, index):
    """page_start/page_end from children pages (monotonic, but min/max is defensive)."""
    for sec in sections:
        pages = [index[cid]["page"] for cid in sec["children"]]
        sec["page_start"] = min(pages) if pages else None
        sec["page_end"] = max(pages) if pages else None


def _assemble_tree(sections, page_count, first_doc_title):
    """L0 root holding the L1 sections. Root title = first doc_title (or null)."""
    return {
        "section_id": "root",
        "title": first_doc_title[1] if first_doc_title else None,
        "title_id": first_doc_title[0] if first_doc_title else None,
        "level": 0,
        "page_start": 0,
        "page_end": max(page_count - 1, 0),
        "children": sections,
    }


def build_dom(doc_ocr, page_dims, doc_id=None, pdf_name=None, dpi=DPI):
    """doc_ocr {page_N: [{label, box, dim7, text, ocr_error, ...}, ...]} +
    page_dims {page_index: (width_px, height_px)} -> doc_dom dict
    {header, index, chrome, tree}."""
    stream = _ordered_stream(doc_ocr)
    caption_of = _detect_captions(stream)
    index, chrome = _route_atoms(stream, caption_of)
    toc_atom_ids = _detect_toc(stream)

    sections, opened_real, first_doc_title = _build_sections(stream, index, caption_of)
    if opened_real == 0:
        sections = _sections_per_page(stream, index)
    _fill_section_pages(sections, index)

    page_count = len(page_dims)
    tree = _assemble_tree(sections, page_count, first_doc_title)

    pages = [{"index": p, "width": w, "height": h}
             for p, (w, h) in sorted(page_dims.items())]
    n_visual = sum(1 for a in index.values() if a["role"] == "visual")
    n_chrome = sum(len(v) for v in chrome.values())

    header = {
        "doc_id": doc_id,
        "schema_version": SCHEMA_VERSION,
        "source": {"doc_ocr": "doc_ocr.json", "pdf": pdf_name},
        "dpi": dpi,
        "coord_space": "raster_px",
        "page_count": page_count,
        "pages": pages,
        "toc_present": bool(toc_atom_ids),
        "toc_atom_ids": toc_atom_ids,
        "stats": {
            "n_atoms": len(index),      # content + visual + captions (chrome excluded)
            "n_sections": len(sections),
            "n_chrome": n_chrome,
            "n_visual": n_visual,
        },
    }

    return {"header": header, "index": index, "chrome": chrome, "tree": tree}
